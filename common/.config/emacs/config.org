(use-package eldoc
  :elpaca nil
  :demand t
  :delight eldoc-mode

  :preface
  (defvar my/eldoc-window-side 'bottom
    "Side on which to display the Eldoc documentation window.")

  (defun my/eldoc-display-buffer (buffer alist)
    "Display Eldoc BUFFER according to `my/eldoc-window-side'."
    (or (display-buffer-reuse-window buffer alist)
        (let ((window-sides-vertical nil))
          (display-buffer-in-side-window
           buffer
           (append `((side . ,my/eldoc-window-side)
                     (slot . 0))
                   (if (eq my/eldoc-window-side 'bottom)
                       '((window-height . 8))
                     '((window-width . 80)))
                   alist)))))

  (defun my/elisp-eldoc-symbol-documentation (callback &rest _)
    "Report full documentation for the Emacs Lisp symbol at point."
    (when-let* ((symbol (symbol-at-point))
                (doc (cond
                      ((fboundp symbol) (documentation symbol t))
                      ((boundp symbol) (documentation-property symbol 'variable-documentation t)))))
      (funcall callback doc :thing (symbol-name symbol) :echo 'skip)
      t))

  (defun my/elisp-eldoc-setup ()
    "Add full symbol documentation to Emacs Lisp Eldoc."
    (setq-local eldoc-documentation-strategy
                #'eldoc-documentation-compose)
    (add-hook 'eldoc-documentation-functions
              #'my/elisp-eldoc-symbol-documentation t t))

  (defun my/eldoc-quit-doc-buffer ()
    "Close the displayed Eldoc documentation window."
    (interactive)
    (when-let ((window (get-buffer-window "*eldoc*" t)))
      (quit-window nil window)))

  (defun my/eldoc-toggle-window-side ()
    "Toggle the Eldoc window between the bottom and right side."
    (interactive)
    (setq my/eldoc-window-side
          (if (eq my/eldoc-window-side 'bottom) 'right 'bottom))
    (when-let ((buffer (get-buffer "*eldoc*")))
      (when-let ((window (get-buffer-window buffer t)))
        (delete-window window))
      (display-buffer buffer))
    (message "Eldoc window: %s" my/eldoc-window-side))

  :init
  (add-to-list 'display-buffer-alist
               '("\\*eldoc\\*"
                 my/eldoc-display-buffer))

  :config
  (global-eldoc-mode 1)

  :custom
  (eldoc-idle-delay 0.2)
  (eldoc-echo-area-display-truncation-message nil)
  (eldoc-echo-area-prefer-doc-buffer t)
  (eldoc-echo-area-use-multiline-p 1)
  (eldoc-help-at-pt t)
  (eldoc-documentation-strategy #'eldoc-documentation-compose)

  :bind (:prefix-map eldoc-prefix-map
                     :prefix "C-c d"
                     ("d" . #'eldoc)
                     ("b" . #'eldoc-doc-buffer)
                     ("t" . #'eldoc-print-current-symbol-info)
                     ("q" . #'my/eldoc-quit-doc-buffer)
                     ("w" . #'my/eldoc-toggle-window-side))

  :hook
  (emacs-lisp-mode . my/elisp-eldoc-setup))

(use-package skeleton
  :elpaca nil

  :init
  (defconst slash-escaped-pairs
    '((?\( . ?\))
      (?\[ . ?\])
      (?\{ . ?\}))
    "Opening and closing characters for backslash-escaped pairs.")

  (defun insert-pair-or-slash-escaped-pair (arg)
    "Insert ARG copies of the typed character, pairing slash-escaped delimiters."
    (interactive "p")
    (let ((closing-char (alist-get last-command-event slash-escaped-pairs)))
      (if (and closing-char (eq (char-before) ?\\))
          (skeleton-insert `(nil
                             ,last-command-event
                             _
                             ,(string ?\\ closing-char))
                           nil)
        (self-insert-command arg))))

  :config
  (dolist (opening-char (mapcar #'car slash-escaped-pairs))
    (keymap-global-set (string opening-char) #'insert-pair-or-slash-escaped-pair)))

    "Return a hidden `consult-buffer' source named NAME for buffers in MODES."
    (list
     :hidden t
     :name name
     :category 'buffer
     :narrow narrow
     :face 'consult-buffer
     :history 'buffer-name-history
     :state #'consult--buffer-state

              (consult--buffer-query
               :mode modes
               :exclude consult-buffer-filter
               :as #'buffer-name))))

  (dolist (source my/consult-mode-buffer-sources)
    (setq consult-buffer-sources (delq source consult-buffer-sources)))

        (mapcar (lambda (spec)
