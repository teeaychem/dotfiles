** straight built-ins

Some libraries bundled with Emacs are also released independently.
Packages may depend on release names, and straight may install a copy after emacs has already loaded the bundled.

Prefer every package reported by Emacs as built in, an external version must be explicitly removed.
This block must therefore precede all third-party packages.

(use-package eldoc
  :straight nil
  :demand t
  :delight eldoc-mode

  :preface
  (defvar my/eldoc-window-side 'bottom
    "Side on which to display the Eldoc documentation window.")

  (defun my/eldoc-display-buffer (buffer alist)
    "Display Eldoc BUFFER according to `my/eldoc-window-side'."
    (or (display-buffer-reuse-window buffer alist)
        (let ((window-sides-vertical nil))
          (display-buffer-in-side-window
           buffer
           (append `((side . ,my/eldoc-window-side)
                     (slot . 0))
                   (if (eq my/eldoc-window-side 'bottom)
                       '((window-height . 8))
                     '((window-width . 80)))
                   alist)))))

  (defun my/eldoc-promote-flymake ()
    "Give Flymake diagnostics priority when Flymake provides Eldoc."
    (when (memq #'flymake-eldoc-function eldoc-documentation-functions)
      (setq-local eldoc-documentation-functions
                  (cons #'flymake-eldoc-function
                        (remove #'flymake-eldoc-function
                                eldoc-documentation-functions)))))

  (defun my/elisp-eldoc-symbol-documentation (callback &rest _)
    "Report full documentation for the Emacs Lisp symbol at point."
    (when-let* ((symbol (symbol-at-point))
                (doc (cond
                      ((fboundp symbol) (documentation symbol t))
                      ((boundp symbol)
                       (documentation-property
                        symbol 'variable-documentation t)))))
      (funcall callback doc
               :thing (symbol-name symbol)
               :echo 'skip)
      t))

  (defun my/elisp-eldoc-setup ()
    "Add full symbol documentation to Emacs Lisp Eldoc."
    (setq-local eldoc-documentation-strategy
                #'eldoc-documentation-compose)
    (add-hook 'eldoc-documentation-functions
              #'my/elisp-eldoc-symbol-documentation t t))

  (defun my/eldoc-quit-doc-buffer ()
    "Close the displayed Eldoc documentation window."
    (interactive)
    (when-let ((window (get-buffer-window "*eldoc*" t)))
      (quit-window nil window)))

  (defun my/eldoc-toggle-window-side ()
    "Toggle the Eldoc window between the bottom and right side."
    (interactive)
    (setq my/eldoc-window-side
          (if (eq my/eldoc-window-side 'bottom) 'right 'bottom))
    (when-let ((buffer (get-buffer "*eldoc*")))
      (when-let ((window (get-buffer-window buffer t)))
        (delete-window window))
      (display-buffer buffer))
    (message "Eldoc window: %s" my/eldoc-window-side))

  :init
  (add-to-list 'display-buffer-alist
               '("\\*eldoc\\*"
                 my/eldoc-display-buffer))

  :config
  (global-eldoc-mode 1)

  :custom
  (eldoc-idle-delay 0.2)
  (eldoc-echo-area-display-truncation-message nil)
  (eldoc-echo-area-prefer-doc-buffer t)
  (eldoc-echo-area-use-multiline-p 1)
  (eldoc-documentation-strategy #'eldoc-documentation-compose)

  :bind (:prefix-map eldoc-prefix-map

                     ("b" . #'eldoc-doc-buffer)
                     ("t" . #'eldoc-print-current-symbol-info)

(use-package skeleton
  :init
  (defconst slash-escaped-pairs
    '((?\( . ?\))
      (?\[ . ?\])
      (?\{ . ?\}))
    "Opening and closing characters for backslash-escaped pairs.")

  (defun insert-pair-or-slash-escaped-pair (arg)
    "Insert ARG copies of the typed character, pairing slash-escaped delimiters."
    (interactive "p")
    (let ((closing-char (alist-get last-command-event slash-escaped-pairs)))
      (if (and closing-char (eq (char-before) ?\\))
          (skeleton-insert `(nil
                             ,last-command-event
                             _
                             ,(string ?\\ closing-char))
                           nil)
        (self-insert-command arg))))

  :config
  (dolist (opening-char (mapcar #'car slash-escaped-pairs))
    (keymap-global-set (string opening-char) #'insert-pair-or-slash-escaped-pair)))

         (file-org (expand-file-name (concat name ".org") user-emacs-directory)))

(use-package rust-ts-mode
  :straight nil

  :mode ("\\.rs\\'" . rust-ts-mode)

  :init
  (defun my/rust-set-edition ()
    "Set `rust-edition' from the current crate's Cargo metadata."
    ;; https://github.com/radian-software/apheleia/issues/278
    (when-let*
        ((cratedir
          (locate-dominating-file default-directory "Cargo.toml"))
         (manifest-path
          (expand-file-name "Cargo.toml" cratedir))
         (env process-environment)
         (path exec-path))
      (setq-local rust-edition
                  (with-temp-buffer
                    (setq-local process-environment env)
                    (setq-local exec-path path)
                    (let ((retcode
                           (process-file
                            "cargo" nil (list (current-buffer) nil) nil
                            "metadata" "--format-version" "1")))
                      (when (/= retcode 0)
                        (error "Cargo metadata --format-version 1 returned %s: %s"
                               retcode (buffer-string)))
                      (goto-char 0)
                      (let ((metadata (json-parse-buffer
                                       :object-type 'alist
                                       :array-type 'list)))
                        (cdr (assq 'edition
                                   (--first
                                    (string= (cdr (assq 'manifest_path it))
                                             manifest-path)
                                    (cdr (assq 'packages metadata)))))))))))

  :hook
  (rust-ts-mode . my/rust-set-edition))

  (defun tempel-setup-capf () ;; Setup completion at point
    ;; Expand exact template names first.  When Eglot owns CAPF, merge
    ;; Tempel completion into Eglot's CAPF so both candidate sets appear.
    (let ((capfs completion-at-point-functions))
      (dolist (capf '(tempel-expand tempel-complete my/tempel-eglot-capf))

      (setq-local completion-at-point-functions
                  (cons #'tempel-expand
                        (if (memq #'eglot-completion-at-point capfs)
                            (cons #'my/tempel-eglot-capf
                                  (remove #'eglot-completion-at-point capfs))
                          (cons #'tempel-complete capfs))))))

  :custom
  (tempel-path (list (concat user-emacs-directory "tempel/common.eld")
                     (concat user-emacs-directory "tempel/modes/*.eld")))
